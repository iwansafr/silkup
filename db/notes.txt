SELECT member.id, member.name, JSON_VALUE(@member.param, "$.nama") AS nama, JSON_VALUE(@member.param,"$.lpk_id") AS lpk, JSON_VALUE(@member.param, "$.alamat") AS alamat, user.username, JSON_VALUE(@member.param,"$.email") AS email, JSON_VALUE(@member.param,"$.user_role_id") AS user_role_id FROM member INNER JOIN user ON(member.user_id = user.id) WHERE ( JSON_VALUE(@member.param,"$.user_role_id") = 3)
